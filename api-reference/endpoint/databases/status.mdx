api: "GET https://api.squarecloud.app/v2/databases/{databaseId}/status"
authMethod: "none"
---

## Request

### Headers

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The API key for your account. You can find this in your [account settings](https://squarecloud.app/account/security).
</ParamField>

### Path

<ParamField path="databaseId" type="string" placeholder="Database ID" required>
  The ID of the database.
</ParamField>

## Response

<ResponseField name="status" type="string">
  Indicates whether the call was successful. `success` if successful, `error` if not.
</ResponseField>
<ResponseField name="response" type="object">

The contents of the response.
<Expandable title="Toggle object">
<ResponseField name="cpu" type="string">
  The current CPU usage percentage. 
</ResponseField>
<ResponseField name="ram" type="string">
  The current RAM usage 
</ResponseField>
<ResponseField name="running" type="boolean">
  The boolean value if is running.
</ResponseField>
<ResponseField name="storage" type="string">
  The current storage it is using.
</ResponseField>
<ResponseField name="network" type="object">
  The network usage and total used.
<Expandable title="Toggle object">
<ResponseField name="total" type="string">
  The total used.
</ResponseField>
<ResponseField name="now" type="string">
  The current use.
</ResponseField>
</Expandable>
</ResponseField>
<ResponseField name="uptime" type="int">
  The timestamp representing the uptime.
</ResponseField>
</Expandable>
</ResponseField>
